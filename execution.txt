# CS01 - CV Creation using LLMs
# Install dependencies:
#   pip install -r requirements.txt
#
# Install and start Ollama, then pull a model:
#   ollama pull llama3
#
# Run with a text profile:
python main.py --input profile.txt --output my_cv.docx --api-key YOUR_GEMINI_API_KEY

# Run with a PDF profile:
python main.py --input profile.pdf --output my_cv.docx --api-key YOUR_GEMINI_API_KEY

# Run with job-description tailoring:
python main.py --input profile.txt --output tailored_cv.docx --job-desc job.txt --api-key YOUR_GEMINI_API_KEY

# Use a different Ollama model:
python main.py --input profile.txt --output my_cv.docx --ollama-model gemma3 --api-key YOUR_GEMINI_API_KEY

# Use environment variable for API key:
export GOOGLE_API_KEY=YOUR_GEMINI_API_KEY
python main.py --input profile.txt --output my_cv.docx
